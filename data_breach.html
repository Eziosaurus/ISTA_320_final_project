data_breach <- read.csv("data/DataBreaches(2004-2021).csv")

For my first graph I decided to do a line plot of total loss in records per year. We can see that 2013 and 2019 had the highest loss in records due to data breaches.

# First Graph
data_breach %>%
  group_by(Year) %>%
  summarize(tot_lost = sum(Records)) %>%
  filter(Year != "2014 and 2015", Year != "2018-2019") %>% 
  ggplot(aes(x = Year,
             y = tot_lost,
             group = 1)) +
  geom_line() +
  scale_y_continuous(labels = label_number(suffix = " B", scale = 1e-9)) +
  ylab("Total Loss in Records") +
  labs(title = "Plot of total loss in records per year")

For my second graph I decided to do a bar plot showing the total loss in records by Organization type. The organization type that has lost the most in data breaches are companies who deal with web services.

# Second Graph
data_breach %>% 
  group_by(Organization.type) %>% 
  summarize(tot_lost = sum(Records)) %>%
  slice_max(order_by = tot_lost, n = 10) %>%
  ggplot(aes(fct_rev(fct_reorder(Organization.type, tot_lost)),
             tot_lost,
             fill = Organization.type)) + 
  geom_bar(stat = "identity") +
  theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
  scale_y_continuous(labels = label_number(suffix = " B", scale = 1e-9)) +
  xlab("Organization Type") +
  ylab("Total Loss in Records") +
  labs(title = "Plot of total loss in records by Organization type",
       fill = "Organization Type")

For my third plot I did another box plot which shows the percentage of which data breach methods were used the most. The most used method is hacking which is no big surprise.

# Third Graph
# Assigning new Data Frame for percentages
method_percentage <- data_breach %>%
  filter(!is.na(Method)) %>%
  group_by(Method) %>%
  summarize(n = n()) %>%
  mutate(total = sum(n),
         percentage = n/total,
         Method = factor(Method))

# Creating bar plot
method_percentage %>%
  slice_max(order_by = percentage, n = 5) %>%
  ggplot(aes(fct_rev(fct_reorder(Method, percentage)),
             percentage,
             fill = Method)) +
  geom_col(position = "dodge") +
  theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
  xlab("Organization Type") +
  ylab("Percentage") +
  labs(title = "Plot of percentage of which data breach methods used",
       fill = "Data Breach Method")
